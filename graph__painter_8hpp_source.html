166  static CanvasInfo buildCanvasInfo(Point2IDGraph& map, double ppm, double margin)
167  {
168  CanvasInfo info;
169  info.ppm = ppm;
170  info.margin = margin;
171  if (map.countNodes() > 0)
172  {
173  Point2 box_min = map.getHeadNode()->data, box_max = map.getHeadNode()->data;
174  for (Point2IDGraph::NodeItr n = map.getHeadNode(); n != map.getTailNode(); n++)
175  {
176  if (n->data.x < box_min.x) box_min.x = n->data.x;
177  if (n->data.y < box_min.y) box_min.y = n->data.y;
178  if (n->data.x > box_max.x) box_max.x = n->data.x;
179  if (n->data.y > box_max.y) box_max.y = n->data.y;
180  }
181  info.box_m = cv::Rect2d(box_min, box_max);
182 
183  info.width = static_cast<int>((box_max.x - box_min.x + 4 * margin) * ppm + 0.5); // + 0.5: Rounding
184  info.height = static_cast<int>((box_max.y - box_min.y + 4 * margin) * ppm + 0.5); // + 0.5: Rounding
185 
186  int margin_p = static_cast<int>(info.margin * info.ppm + 0.5); // + 0.5: Rounding
187  info.box_p.x = margin_p;
188  info.box_p.y = margin_p;
189  info.box_p.width = info.width - 2 * margin_p;
190  info.box_p.height = info.height - 2 * margin_p;
191  }
192  return info;
193  }

225  static bool drawGrid(cv::Mat& image, const CanvasInfo& info, double grid_step, const cv::Vec3b& color, int thickness = 1, double unit_font_scale = 0.5, const cv::Vec3b& unit_color = cx::COLOR_BLACK, const cv::Point& unit_pos = cv::Point(100, 10))
226  {
227  CV_DbgAssert(!image.empty());
228  if (thickness <= 0) return false;
229 
230  if (grid_step < 0)
231  {
232  // Select cell size automatically
233  double sz = std::max(info.box_m.width, info.box_m.height);
234  grid_step = pow(10., int(log10(sz / 4)));
235  if (grid_step < 1) grid_step = 1;
236  else if (grid_step > 1000) grid_step = 1000;
237  }
238 
239  Point2 center_map(info.box_m.x + info.box_m.width / 2, info.box_m.y + info.box_m.height / 2);
240  Point2 center_grid = cvtMeter2Pixel(Point2(int(center_map.x / grid_step) * grid_step, int(center_map.y / grid_step) * grid_step), info);
241  for (int i = 0;; i++)
242  {
243  double y = center_grid.y - grid_step * i * info.ppm;
244  if (y < info.box_p.tl().y || y > info.box_p.br().y) break;
245  cv::line(image, cv::Point2d(info.box_p.tl().x, y), cv::Point2d(info.box_p.br().x, y), color, thickness); // Upward X-directional grids
246  }
247  for (int i = 1;; i++)
248  {
249  double y = center_grid.y + grid_step * i * info.ppm;
250  if (y < info.box_p.tl().y || y > info.box_p.br().y) break;
251  cv::line(image, cv::Point2d(info.box_p.tl().x, y), cv::Point2d(info.box_p.br().x, y), color, thickness); // Downward X-directional grids
252  }
253  for (int i = 0;; i++)
254  {
255  double x = center_grid.x + grid_step * i * info.ppm;
256  if (x < info.box_p.tl().x || x > info.box_p.br().x) break;
257  cv::line(image, cv::Point2d(x, info.box_p.tl().y), cv::Point2d(x, info.box_p.br().y), color, thickness); // Rightward Y-directional grids
258  }
259  for (int i = 1;; i++)
260  {
261  double x = center_grid.x - grid_step * i * info.ppm;
262  if (x < info.box_p.tl().x || x > info.box_p.br().x) break;
263  cv::line(image, cv::Point2d(x, info.box_p.tl().y), cv::Point2d(x, info.box_p.br().y), color, thickness); // Leftward Y-directional grids
264  }
265  if (unit_font_scale > 0)
266  cv::putText(image, cv::format("Grid: %d [m]", int(grid_step)), cv::Point(image.cols, image.rows) - unit_pos, cv::FONT_HERSHEY_DUPLEX, unit_font_scale, unit_color);
267  return true;
268  }

301  static bool drawEdge(cv::Mat& image, const CanvasInfo& info, const Point2ID& from, const Point2ID& to, double radius, const cv::Vec3b& color, int thickness = 1, double arrow_length = -1)
302  {
303  CV_DbgAssert(!image.empty());
304  if (thickness <= 0) return false;
305 
306  const double r = radius * info.ppm;
307  const double a = arrow_length * info.ppm;
308 
309  // Draw an edge
310  Point2 p = cvtMeter2Pixel(from, info);
311  Point2 q = cvtMeter2Pixel(to, info);
312  double theta = atan2(q.y - p.y, q.x - p.x);
313  Point2 delta(r * cos(theta), r * sin(theta));
314  p = p + delta;
315  q = q - delta;
316  cv::line(image, p, q, color, thickness);
317 
318  // Draw its arrow
319  if (a > 0)
320  {
321  double theta_p = theta + CV_PI / 6;
322  double theta_m = theta - CV_PI / 6;
323  cv::line(image, q, q - a * Point2(cos(theta_p), sin(theta_p)), color, thickness);
324  cv::line(image, q, q - a * Point2(cos(theta_m), sin(theta_m)), color, thickness);
325  }
326  return true;
327  }

330  {
331  CV_DbgAssert(!image.empty());

333

346

351  double theta_m = theta - CV_PI / 6;
